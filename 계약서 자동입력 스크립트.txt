/**
 * 일반 전화번호 포맷팅 (02-000-0000 등)
 */
function formatTel(no) {
  if (!no) return "";
  const clean = no.replace(/\D/g, "");
  
  // 서울(02) 번호 처리
  if (clean.startsWith("02")) {
    if (clean.length === 9) return clean.replace(/(\d{2})(\d{3})(\d{4})/, "$1-$2-$3");
    if (clean.length === 10) return clean.replace(/(\d{2})(\d{4})(\d{4})/, "$1-$2-$3");
  }
  
  // 일반 지역번호 및 인터넷 전화
  if (clean.length === 10) return clean.replace(/(\d{3})(\d{3})(\d{4})/, "$1-$2-$3");
  if (clean.length === 11) return clean.replace(/(\d{3})(\d{4})(\d{4})/, "$1-$2-$3");
  
  return no; // 형식이 맞지 않으면 원본 반환
}

function formatMobile(no) {
  if (!no) return "";
  const clean = no.replace(/\D/g, "");
  if (clean.length === 11) return clean.replace(/(\d{3})(\d{4})(\d{4})/, "$1-$2-$3");
  if (clean.length === 10) return clean.replace(/(\d{3})(\d{3})(\d{4})/, "$1-$2-$3");
  return no;
}

// [callGeminiVisionAPI, correctKoreanAddress, formatBizNo는 기존과 동일하게 유지]
function callGeminiVisionAPI(base64Data, mimeType) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-flash-latest:generateContent?key=${GEMINI_API_KEY}`;
  const promptText = `Analyze this contract image. The layout may vary between different versions. 
    Find and extract the following fields. Return ONLY raw JSON.
    Fields: bizNo, clinicNo, clinicName, ownerName, address, tel, mobile, email, specialTerms, contractDate, promoCid, promoNaver, promoSurvey, promoZeroEighty.
    * Requirements: Extract EXACT TEXT for promo fields. If clinicNo is present, extract all digits. Output must be a valid JSON string.`;

  const payload = { contents: [{ parts: [{ text: promptText }, { inline_data: { mime_type: mimeType, data: base64Data } }] }] };
  const options = { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload), muteHttpExceptions: true };

  try {
    const response = UrlFetchApp.fetch(url, options);
    const resText = response.getContentText();
    const json = JSON.parse(resText);
    if (json.candidates && json.candidates.length > 0 && json.candidates[0].content) {
      let aiText = json.candidates[0].content.parts[0].text;
      const jsonMatch = aiText.match(/\{[\s\S]*\}/);
      if (jsonMatch) return JSON.parse(jsonMatch[0]);
    }
    return null;
  } catch (e) { return null; }
}

function correctKoreanAddress(rawAddress) {
  if (!rawAddress) return { sido: "", sigungu: "", fullAddress: "" };
  let cleanAddr = rawAddress.trim();
  let parts = cleanAddr.split(/\s+/);
  let sido = parts[0] || "";
  let sigungu = parts.length > 1 ? parts[1] : "";
  const sidoMap = {"서울특별시": "서울", "경기도": "경기", "강원도": "강원", "충청북도": "충북", "충청남도": "충남", "전라북도": "전북", "전라남도": "전남", "경상북도": "경북", "경상남도": "경남", "제주도": "제주", "인천광역시": "인천", "부산광역시": "부산", "대전광역시": "대전", "대구광역시": "대구", "울산광역시": "울산", "광주광역시": "광주"};
  sido = sidoMap[sido] || (sido.length > 2 ? sido.substring(0, 2) : sido);
  return { sido: sido, sigungu: sigungu, fullAddress: cleanAddr };
}

function formatBizNo(no) { if (!no) return ""; const clean = no.replace(/\D/g, ""); return (clean.length === 10) ? clean.replace(/(\d{3})(\d{2})(\d{5})/, "$1-$2-$3") : no; }
